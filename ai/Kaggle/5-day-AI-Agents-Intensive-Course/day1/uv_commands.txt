uv pip install -r requirements.txt

uv pip compile pyproject.toml -o requirements.txt -- Generate a lock file for reproducible builds
uv pip sync requirements.txt -- Sync environment with a Lock file (replaces pip-sync)
uv run python kaggle.py -- Running scripts
